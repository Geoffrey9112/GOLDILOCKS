CMAKE_MINIMUM_REQUIRED (VERSION 2.6)
# Avoid warnings in higher versions
IF("${CMAKE_MAJOR_VERSION}.${CMAKE_MINOR_VERSION}" GREATER 2.6)
 CMAKE_POLICY(VERSION 2.8)
ENDIF()

INCLUDE(FindJava.cmake)
INCLUDE(UseJava.cmake)

SET(CMAKE_JAVA_COMPILE_FLAGS -sourcepath dev/src)
SET(CMAKE_JAVA_MANIFEST_FILE META-INF/MANIFEST.MF)
SET(CMAKE_JAVA_RESOURCE_FILE META-INF/services/java.ErrorViewing.Driver)
ADD_JAR(ErrorViewing ${CMAKE_CURRENT_SOURCE_DIR}/dev/bin
  dev/src/*.java)
